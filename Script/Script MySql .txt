CREATE DATABASE WK;
USE WK;

CREATE TABLE Clientes
(
	Codigo INTEGER UNSIGNED NOT NULL AUTO_INCREMENT, 
	Nome VARCHAR(40) NULL, 
	Cidade VARCHAR(20) NULL, 
	UF CHAR(2) NULL, 
	PRIMARY KEY(Codigo)
);

CREATE TABLE Produtos
(
	Codigo INTEGER UNSIGNED NOT NULL AUTO_INCREMENT, 
	Descricao VARCHAR(40) NULL, 
	PrecoVenda FLOAT(8,2) NULL, 
	PRIMARY KEY(Codigo)
);
